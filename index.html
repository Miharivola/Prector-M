function generatePrediction() { const userTime = document.getElementById('input-time').value; const seed = document.getElementById('input-seed').value; if(!userTime || !seed) { alert("Fenoy ny Heure sy ny Graine azafady!"); return; } // 1. KAJY NY LERA (Lojika miampy 1 hatramin'ny 3 minitra kisendrasendra) let parts = userTime.split(':'); let date = new Date(); date.setHours(parseInt(parts[0]), parseInt(parts[1]), parseInt(parts[2])); // Manampy elanelana raikitra nefa misy segondra miovaova date.setMinutes(date.getMinutes() + 2); let secRandom = Math.floor(Math.random() * 60); date.setSeconds(secRandom); const predictionTime = date.toLocaleTimeString('fr-FR', { hour12: false }); // 2. ALGORITHM "PROVABLY REALISTIC" // Mampiasa ny 'seed' sy ny 'lera' ho 'hash' tsotra let hashSeed = 0; for (let i = 0; i < seed.length; i++) { hashSeed += seed.charCodeAt(i); } let combinedFactor = (hashSeed + date.getMinutes() + secRandom) % 100; let baseVal, secuVal, riskVal; // LOJIKA: Mizara ny vintana ho 3 sokajy (toy ny tena lalao) if (combinedFactor < 40) { // 40% Vintana Ratsy (Blue) baseVal = (1.01 + (Math.random() * 0.19)).toFixed(2); secuVal = (1.20 + (Math.random() * 0.50)).toFixed(2); riskVal = (2.00 + (Math.random() * 2.00)).toFixed(2); } else if (combinedFactor < 85) { // 45% Vintana Antonony (Purple) baseVal = (1.20 + (Math.random() * 0.30)).toFixed(2); secuVal = (2.50 + (Math.random() * 4.00)).toFixed(2); riskVal = (8.00 + (Math.random() * 10.00)).toFixed(2); } else { // 15% Vintana Tsara (Rose/High Purple) baseVal = (1.50 + (Math.random() * 0.40)).toFixed(2); secuVal = (6.00 + (Math.random() * 10.00)).toFixed(2); riskVal = (20.00 + (Math.random() * 80.00)).toFixed(2); } // 3. ASEHOY NY VALINY document.getElementById('result-area').style.display = 'block'; document.getElementById('res-hour').innerText = predictionTime; document.getElementById('v1').innerText = "x" + baseVal; document.getElementById('v2').innerText = "x" + secuVal; document.getElementById('v3').innerText = "x" + riskVal; // STATUS TEXT (Miankina amin'ny combinedFactor) const st = document.getElementById('status-text'); if(combinedFactor >= 60) { st.innerHTML = "💛 TOUR HO TSARA"; st.style.color = "#2ecc71"; st.style.backgroundColor = "rgba(46, 204, 113, 0.1)"; } else { st.innerHTML = "💔 TOUR HO RATSY"; st.style.color = "#e74c3c"; st.style.backgroundColor = "rgba(231, 76, 60, 0.1)"; } }
